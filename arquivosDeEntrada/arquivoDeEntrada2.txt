struct Endereco {
    string logradouro;
    string cidade;
    string estado;
    string cep;
};

class Pessoa {
    attribute string nome;
    attribute int idade;
    attribute Endereco endereco;
};

class Disciplina extends Disciplina {
    attribute string nome;
};

class Aluno extends Pessoa {
    attribute string matricula;
    attribute list<Disciplina> disciplinas;
    attribute list<Turma> turmas;
};

class Professor extends Pessoa {
    attribute string codigoFuncionario;
    attribute list<Disciplina> disciplinasLecionadas;
};

class Turma {
    attribute string codigoTurma;
    attribute Professor professor;
    attribute list<Aluno> alunos;
};

class Escola {
    attribute string nome;
    attribute Endereco endereco;
    attribute list<Turma> turmas;
};
